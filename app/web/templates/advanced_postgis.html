Demo 1: ポイントの INSERT と地図表示

    

        
Snowflake 東京オフィス（八重洲2-2-1）の座標を DB に INSERT し、地図上にプロットします。

        
-- Snowflake Tokyo Office を INSERT
INSERT INTO postgis_demo.points (name, category, geom) VALUES
('Snowflake Tokyo Office', 'office',
 ST_SetSRID(ST_MakePoint(139.77076, 35.67986), 4326))
ON CONFLICT (name) DO NOTHING;

-- 地図表示用に座標を取得
SELECT name, category,
       ST_X(geom) AS lng, ST_Y(geom) AS lat
FROM postgis_demo.points WHERE name = 'Snowflake Tokyo Office';

        
        

    







    
Demo 2: 複数ポイントのプロット

    

        
東京駅周辺のランドマークを追加し、地図上に全ポイントをプロットします。

        
-- 東京駅周辺のランドマークを INSERT
INSERT INTO postgis_demo.points (name, category, geom) VALUES
('東京駅', 'station', ST_SetSRID(ST_MakePoint(139.7671, 35.6812), 4326)),
('日本橋', 'landmark', ST_SetSRID(ST_MakePoint(139.7745, 35.6839), 4326)),
('銀座四丁目交差点', 'landmark', ST_SetSRID(ST_MakePoint(139.7649, 35.6714), 4326)),
('皇居前広場', 'park', ST_SetSRID(ST_MakePoint(139.7528, 35.6852), 4326))
ON CONFLICT (name) DO NOTHING;

-- 全ポイントを取得
SELECT name, category,
       ST_X(geom) AS lng, ST_Y(geom) AS lat
FROM postgis_demo.points ORDER BY id;

Demo 4: ポリゴンの包含判定 (ST_Contains)

    

        
八重洲エリアを四角く囲うポリゴンを作成し、ST_Contains で各ポイントが中に入っているかを判定します。

        
-- 八重洲エリアのポリゴンを作成
INSERT INTO postgis_demo.polygons (name, geom) VALUES
('八重洲エリア',
 ST_SetSRID(ST_MakePolygon(ST_GeomFromText(
   'LINESTRING(139.768 35.677, 139.774 35.677, 139.774 35.683, 139.768 35.683, 139.768 35.677)'
 )), 4326))
ON CONFLICT (name) DO NOTHING;

-- 包含判定
SELECT p.name AS point_name,
       ST_Contains(
         (SELECT geom FROM postgis_demo.polygons WHERE name = '八重洲エリア'),
         p.geom
       ) AS inside_yaesu
FROM postgis_demo.points p ORDER BY p.id;
