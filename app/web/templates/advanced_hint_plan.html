-- Check actual row counts / 実際の行数を確認
SELECT status, COUNT(*) AS actual_count
FROM hint_demo.orders GROUP BY status ORDER BY actual_count DESC;

-- Check planner statistics (pg_stats) / プランナーの統計情報（pg_stats）
SELECT most_common_vals, most_common_freqs
FROM pg_stats
WHERE schemaname = 'hint_demo' AND tablename = 'orders' AND attname = 'status';
