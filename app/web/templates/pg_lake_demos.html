Step 1: {{ t('lake.demo4.step1') }}
CREATE SCHEMA IF NOT EXISTS iot;

CREATE TABLE iot.sensor_data (
    ts        TIMESTAMPTZ NOT NULL,
    device_id INT         NOT NULL,
    temp      DOUBLE PRECISION,
    humidity  DOUBLE PRECISION,
    pressure  DOUBLE PRECISION
) PARTITION BY RANGE (ts);

SELECT partman.create_parent(
    p_parent_table := 'iot.sensor_data',
    p_control      := 'ts',
    p_interval     := 'daily',
    p_premake      := 3
);
Step 2: {{ t('lake.demo4.step2') }}
CREATE TABLE iot.sensor_archive (
    ts        TIMESTAMPTZ NOT NULL,
    device_id INT         NOT NULL,
    temp      DOUBLE PRECISION,
    humidity  DOUBLE PRECISION,
    pressure  DOUBLE PRECISION
) USING iceberg
  WITH (partition_by = 'day(ts)');
Step 3: {{ t('lake.demo4.step3_pipe') }}
-- pg_incremental: continuous sync to Iceberg (1-minute interval)
SELECT incremental.create_pipeline(
    pipeline_name := 'iot_to_iceberg',
    query         := $$
        INSERT INTO iot.sensor_archive (ts, device_id, temp, humidity, pressure)
        SELECT ts, device_id, temp, humidity, pressure
        FROM iot.sensor_data
        WHERE ts >= $1 AND ts < $2
    $$,
    interval      := '1 minute'::interval
);

-- Check pipeline status
SELECT * FROM incremental.pipelines;
Step 4: {{ t('lake.demo4.step4_data') }}
INSERT INTO iot.sensor_data (ts, device_id, temp, humidity, pressure)
SELECT
    now() - (random() * interval '60 days'),
    (random() * 100)::int,
    20.0 + random() * 15.0,
    40.0 + random() * 40.0,
    1000.0 + (random() - 0.5) * 50.0
FROM generate_series(1, 300000) AS s(i);
Step 5: {{ t('lake.demo4.step5_sync') }}
-- Manual pipeline execution (in production, pg_cron runs this automatically)
CALL incremental.execute_pipeline('iot_to_iceberg');

-- Verify data synced to Iceberg archive
SELECT date_trunc('day', ts) AS day, count(*) AS rows
FROM iot.sensor_archive
GROUP BY 1 ORDER BY 1;
Step 6: {{ t('lake.demo4.step6_switch') }}

{{ t('lake.demo4.step6_desc') }}

DO $$
DECLARE
    target_date DATE := current_date - 4;
    part_name   TEXT := 'iot.sensor_data_p' || to_char(target_date, 'YYYY_MM_DD');
    fdw_name    TEXT := 'iot.sensor_cold_p' || to_char(target_date, 'YYYY_MM_DD');
    s3_path     TEXT := 's3://your-bucket/iot/sensor_archive/day=' || target_date || '/';
    range_start TIMESTAMPTZ := target_date::timestamptz;
    range_end   TIMESTAMPTZ := (target_date + 1)::timestamptz;
BEGIN
    -- 1. DETACH the heap partition
    EXECUTE format('ALTER TABLE iot.sensor_data DETACH PARTITION %I', part_name);

    -- 2. DROP the heap partition (data already in S3 Iceberg)
    EXECUTE format('DROP TABLE IF EXISTS %I', part_name);

    -- 3. CREATE FOREIGN TABLE pointing to S3 Iceberg data
    EXECUTE format(
        'CREATE FOREIGN TABLE %I (
            ts        TIMESTAMPTZ NOT NULL,
            device_id INT         NOT NULL,
            temp      DOUBLE PRECISION,
            humidity  DOUBLE PRECISION,
            pressure  DOUBLE PRECISION
        ) SERVER pg_lake_server OPTIONS (path %L)',
        fdw_name, s3_path
    );

    -- 4. ATTACH as partition (same range as the original)
    EXECUTE format(
        'ALTER TABLE iot.sensor_data ATTACH PARTITION %I FOR VALUES FROM (%L) TO (%L)',
        fdw_name, range_start, range_end
    );

    RAISE NOTICE 'Switched % from heap to FDW (S3: %)', part_name, s3_path;
END $$;
Step 7: {{ t('lake.demo4.step7_verify') }}
-- A: Yesterday only (heap only, no S3 access)
EXPLAIN (COSTS OFF)
SELECT avg(temp), avg(humidity) FROM iot.sensor_data
WHERE ts >= now() - interval '1 day';
-- B: This week (heap + some Cold)
SELECT date_trunc('day', ts) AS day, count(*), avg(temp)::numeric(5,2) AS avg_temp
FROM iot.sensor_data WHERE ts >= now() - interval '7 days'
GROUP BY 1 ORDER BY 1;
-- C: This month (heap + ~27 Cold partitions)
SELECT date_trunc('day', ts) AS day, count(*), avg(temp)::numeric(5,2) AS avg_temp
FROM iot.sensor_data WHERE ts >= date_trunc('month', now())
GROUP BY 1 ORDER BY 1;
-- D: Full range (all partitions: heap + FDW)
SELECT date_trunc('week', ts) AS week, count(*), avg(temp)::numeric(5,2) AS avg_temp
FROM iot.sensor_data
GROUP BY 1 ORDER BY 1;
Step 8: {{ t('lake.demo4.step8_update') }}
-- Update historical data via Iceberg archive (not via FDW partition)
UPDATE iot.sensor_archive
SET temp = 25.0
WHERE device_id = 42 AND ts::date = current_date - 10
LIMIT 1;

-- === Snowflake: Read Iceberg table from S3 ===

-- 1. External Volume
CREATE OR REPLACE EXTERNAL VOLUME pg_lake_vol
    STORAGE_LOCATIONS = ((
        NAME = 'pg_lake_s3',
        STORAGE_BASE_URL = 's3://your-bucket/pg_lake/',
        STORAGE_PROVIDER = 'S3',
        STORAGE_AWS_ROLE_ARN = 'arn:aws:iam::123456789012:role/pg-lake-access'
    ));

-- 2. Catalog Integration
CREATE OR REPLACE CATALOG INTEGRATION pg_lake_catalog
    CATALOG_SOURCE = OBJECT_STORE
    TABLE_FORMAT = ICEBERG
    ENABLED = TRUE;

-- 3. Register Iceberg Table
CREATE OR REPLACE ICEBERG TABLE analytics.sensor_from_pg
    EXTERNAL_VOLUME = 'pg_lake_vol'
    CATALOG = 'pg_lake_catalog'
    METADATA_FILE_PATH = 'sensor_archive/metadata/v1.metadata.json';

-- 4. Query
SELECT date_trunc('hour', ts) AS hour,
       avg(temp) AS avg_temp, count(*) AS readings
FROM analytics.sensor_from_pg
GROUP BY 1 ORDER BY 1;

-- 5. Refresh metadata
ALTER ICEBERG TABLE analytics.sensor_from_pg REFRESH;
