{{ t('lake.overview') }}
  • CREATE TABLE ... USING iceberg {% if lang == 'ja' %}で Iceberg テーブルを作成{% else %}to create Iceberg tables{% endif %}
  • {% if lang == 'ja' %}通常の INSERT / UPDATE / DELETE がそのまま動作{% else %}Standard INSERT / UPDATE / DELETE works as-is{% endif %}
  • {% if lang == 'ja' %}S3 上の Parquet / CSV / JSON を Foreign Table で直接参照{% else %}Query Parquet / CSV / JSON on S3 via Foreign Tables{% endif %}
  • COPY TO/FROM 's3://...' {% if lang == 'ja' %}でデータレイク読み書き{% else %}for data lake read/write{% endif %}
  • {% if lang == 'ja' %}パーティショニング{% else %}Partitioning{% endif %}: day(), month(), bucket()

-- Create Iceberg table / Iceberg テーブル作成
CREATE TABLE events (
  event_time timestamptz,
  user_id int,
  data text
) USING iceberg
  WITH (partition_by = 'day(event_time)');

-- Standard DML / 通常の DML
INSERT INTO events VALUES (now(), 1, 'hello');

-- Export to S3 / S3 へエクスポート
COPY events TO 's3://bucket/events.parquet';

-- Query Parquet on S3 via foreign table
-- S3 の Parquet を外部テーブルとして参照
CREATE FOREIGN TABLE ext_data ()
  SERVER pg_lake
  OPTIONS (path 's3://bucket/*.parquet');
